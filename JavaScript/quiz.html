var a = 1; var result = (function(x) { return function(y) { var a = 2; return (function(z) { return function(w) { return a + x + y + z + w; }; })(y * 2); }; })(a * 3)(a * 4); console.log(result(5));

输出结果：

控制台输出:
18

深度解析：

这个题目展示了极其复杂的多层IIFE与闭包嵌套：

  1. 全局变量 a = 1
  2. 第一层IIFE：参数 x = a * 3 = 3
  3. 返回一个函数，接受参数 y = a * 4 = 4
  4. 在返回的函数内部：var a = 2（屏蔽了全局的a）
  5. 第二层IIFE：参数 z = y * 2 = 8
  6. 返回一个函数，接受参数 w = 5
  7. 最终计算：a(2) + x(3) + y(4) + z(8) + w(5) = 2 + 3 + 4 + 8 + 5 = 22
注意： 这里有一个关键点 - 在计算y时使用的是全局的a(1)，而不是内部声明的a(2)，因为参数传递发生在函数内部变量声明之前。
实际上，由于JavaScript的函数作用域和变量提升机制，内部var a = 2的声明会提升到函数顶部，但赋值发生在原地。参数y的计算发生在函数执行时，此时内部a的赋值还未发生，所以y = a * 4 = 1 * 4 = 4。

深度解析：

这个题目结合了IIFE与JavaScript事件循环的复杂机制：

  1. 同步代码执行：输出1、10
  2. 微任务队列（Promise.then）优先于宏任务队列（setTimeout）执行
  3. Promise.then中的IIFE立即执行，输出6、7、9
  4. IIFE内部的Promise.then创建新的微任务，输出8

var x = 10; function outer() { var x = 20; return (function() { var x = 30; return { getX: function() { return x; }, setX: function(val) { x = val; }, getOuterX: function() { return outer.x; }, getGlobalX: function() { return this.x; }.bind(this) }; })(); } var obj1 = outer(); var obj2 = outer(); console.log(obj1.getX()); console.log(obj2.getX()); obj1.setX(40); console.log(obj1.getX()); console.log(obj2.getX()); console.log(obj1.getOuterX()); console.log(obj1.getGlobalX()); obj1.getGlobalX = obj1.getGlobalX.bind({x: 50}); console.log(obj1.getGlobalX());

深度解析：

这是所有概念的终极融合：

  1. 全局x = 10
  2. outer函数内部x = 20
  3. IIFE内部x = 30，返回一个对象
  4. obj1和obj2是两个独立的闭包实例，各自有独立的x
  5. 初始时obj1和obj2的getX都返回30

🚀 题目六：极限递归与IIFE的复杂交互

var result = (function f(n) { return n > 1 ? n * f(n - 1) : 1; })(5); console.log(result); var factorial = (function() { var cache = {}; return function(n) { if (n in cache) { console.log('From cache:', n); return cache[n]; } if (n <= 1) { return 1; } var result = n * factorial(n - 1); cache[n] = result; console.log('Calculated:', n); return result; }; })(); console.log(factorial(4)); console.log(factorial(5));

深度解析：

这个题目展示了IIFE与递归的复杂交互：

  1. 第一个IIFE是递归计算阶乘的立即执行版本，计算5! = 120
  2. 第二个是带缓存的阶乘函数，使用闭包保存cache对象
